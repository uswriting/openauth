<Section type="about">
Use this provider to authenticate with Facebook. Supports both OAuth2 and OIDC.

#### Using OAuth

```ts {5-8}
import { FacebookProvider } from "@uswriting/openauth/provider/facebook"

export default issuer({
  providers: {
    facebook: FacebookProvider({
      clientId: "1234567890",
      clientSecret: "0987654321"
    })
  }
})
```

#### Using OIDC

```ts {5-7}
import { FacebookOidcProvider } from "@uswriting/openauth/provider/facebook"

export default issuer({
  providers: {
    facebook: FacebookOidcProvider({
      clientId: "1234567890"
    })
  }
})
```
</Section>
